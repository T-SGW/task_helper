CMDで以下を実行
mvn archetype:generate -DgroupId=com.example -DartifactId=JavaFXTest -DarchetypeArtifactId=maven-archetype-quickstart -DinteractiveMode=false



POMに以下を追記
<dependencies>
    <dependency>
        <groupId>org.openjfx</groupId>
        <artifactId>javafx-controls</artifactId>
        <version>20.0.2</version>
    </dependency>
    <dependency>
        <groupId>org.openjfx</groupId>
        <artifactId>javafx-fxml</artifactId>
        <version>20.0.2</version>
    </dependency>
</dependencies>

<properties>
    <maven.compiler.source>17</maven.compiler.source>
    <maven.compiler.target>17</maven.compiler.target>
</properties>



サンプルコードを作成
package com.example;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.Label;
import javafx.stage.Stage;

public class App extends Application {
    @Override
    public void start(Stage stage) {
        Label label = new Label("JavaFX is working!");
        Scene scene = new Scene(label, 300, 100);
        stage.setScene(scene);
        stage.setTitle("JavaFX Test");
        stage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }
}


Mavenでビルド実行
mvn clean compile exec:java -Dexec.mainClass="com.example.App" -Dexec.args="--module-path <javafx_lib_path> --add-modules javafx.controls"
